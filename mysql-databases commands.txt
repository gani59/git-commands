#command to take the database backup(it can pronounce as "dump" in mysql ) 
  (syntax:- sudo mysqldump -u username -p passwod databasename > /opt/directory-name/database-name.sql)
 *e.g;- sudo mysqldump -u root -p mysql > /opt/mysqlbackups/mysql.sql
